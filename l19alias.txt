=>Alias

git status
git config --global alias.st "status"
or
git config --global alias.st status

git st
git status
git config --global alias.st

git branch
git config --global alias.brh "branch"
git brh
git branch
git config --global alias.brh

git commit -m ""
git config --global alias.cm "commit"
git cm
git commit
git config --global alias.cm
